1.Find the count of movies released in each year.
db.movies.aggregate([
  {
    $group: {
      _id: { $year: "$released" },
      count: { $sum: 1 } 
    }
  }
])

    $match: { released: { $exists: true } } // Filter for documents with release_date
  }
])
